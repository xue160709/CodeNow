# 同步语音合成（WebSocket T2A）

基于文本的流式语音合成。单次文本长度上限以平台文档为准（常见说明为约 10,000 字符）。

- **连接**：`wss://api.minimaxi.com/ws/v1/t2a_v2`  
- **鉴权**：握手请求头 `Authorization: Bearer <MINIMAX_API_KEY>`

## 支持模型

| 模型 | 说明 |
| :--- | :--- |
| speech-2.8-hd | 高自然度 |
| speech-2.8-turbo | 高速 |
| speech-2.6-hd | 低延迟 |
| speech-2.6-turbo | 极速，适合对话 / 数字人 |
| speech-02-hd | 韵律与复刻相似度 |
| speech-02-turbo | 高速，小语种加强 |

## 支持语言（节选）

共约 40 种语言，包括但不限于：中文、粤语、英语、西班牙语、法语、俄语、德语、葡萄牙语、阿拉伯语、意大利语、日语、韩语、印尼语、越南语、土耳其语、荷兰语、乌克兰语、泰语、波兰语、罗马尼亚语、希腊语、捷克语、芬兰语、印地语、保加利亚语、丹麦语、希伯来语、马来语、波斯语、斯洛伐克语、瑞典语、克罗地亚语、菲律宾语、匈牙利语、挪威语、斯洛文尼亚语、加泰罗尼亚语、尼诺斯克语、泰米尔语、阿非利卡语等。完整列表以官方文档为准。

## 流式示例（Python）

依赖：`pip install websockets`。若示例使用 `mpv` 实时播放，需本机已安装 [mpv](https://mpv.io/installation/)。

```python
import asyncio
import json
import os
import ssl
import subprocess

import websockets

MODEL = "speech-2.8-hd"
FILE_FORMAT = "mp3"


class StreamAudioPlayer:
    def __init__(self):
        self.mpv_process = None

    def start_mpv(self):
        try:
            self.mpv_process = subprocess.Popen(
                ["mpv", "--no-cache", "--no-terminal", "--", "fd://0"],
                stdin=subprocess.PIPE,
                stdout=subprocess.DEVNULL,
                stderr=subprocess.DEVNULL,
            )
            return True
        except FileNotFoundError:
            print("mpv not found")
            return False

    def play_audio_chunk(self, hex_audio: str) -> bool:
        if self.mpv_process and self.mpv_process.stdin:
            self.mpv_process.stdin.write(bytes.fromhex(hex_audio))
            self.mpv_process.stdin.flush()
            return True
        return False

    def stop(self):
        if self.mpv_process:
            if self.mpv_process.stdin and not self.mpv_process.stdin.closed:
                self.mpv_process.stdin.close()
            try:
                self.mpv_process.wait(timeout=20)
            except subprocess.TimeoutExpired:
                self.mpv_process.terminate()


async def establish_connection(api_key: str):
    url = "wss://api.minimaxi.com/ws/v1/t2a_v2"
    hdrs = {"Authorization": f"Bearer {api_key}"}
    ssl_context = ssl.create_default_context()
    # 示例关闭校验；生产环境建议开启证书校验并按平台要求配置
    ssl_context.check_hostname = False
    ssl_context.verify_mode = ssl.CERT_NONE

    ws = await websockets.connect(url, additional_headers=hdrs, ssl=ssl_context)
    connected = json.loads(await ws.recv())
    if connected.get("event") == "connected_success":
        return ws
    return None


async def start_task(ws):
    start_msg = {
        "event": "task_start",
        "model": MODEL,
        "voice_setting": {
            "voice_id": "male-qn-qingse",
            "speed": 1,
            "vol": 1,
            "pitch": 0,
            "english_normalization": False,
        },
        "audio_setting": {
            "sample_rate": 32000,
            "bitrate": 128000,
            "format": FILE_FORMAT,
            "channel": 1,
        },
    }
    await ws.send(json.dumps(start_msg))
    response = json.loads(await ws.recv())
    return response.get("event") == "task_started"


async def continue_task_with_stream_play(ws, text: str, player: StreamAudioPlayer):
    await ws.send(json.dumps({"event": "task_continue", "text": text}))

    chunk_counter = 1
    audio_data = b""

    while True:
        try:
            response = json.loads(await ws.recv())
        except Exception:
            break

        if "data" in response and "audio" in response["data"]:
            audio_hex = response["data"]["audio"]
            if audio_hex and player.play_audio_chunk(audio_hex):
                audio_bytes = bytes.fromhex(audio_hex)
                audio_data += audio_bytes
                print(f"chunk #{chunk_counter}")
                chunk_counter += 1

        if response.get("is_final"):
            if player.mpv_process and player.mpv_process.stdin:
                player.mpv_process.stdin.close()
            out = f"output.{FILE_FORMAT}"
            with open(out, "wb") as f:
                f.write(audio_data)
            print(f"saved {out}, chunks={chunk_counter - 1}")
            estimated_duration = len(audio_data) * 0.0625 / 1000
            return max(estimated_duration + 5, 10)

    return 10


async def close_connection(ws):
    if ws:
        try:
            await ws.send(json.dumps({"event": "task_finish"}))
            await ws.close()
        except Exception:
            pass


async def main():
    api_key = os.environ["MINIMAX_API_KEY"]
    text = "这是一段用于测试语音合成的示例文本。"

    player = StreamAudioPlayer()
    if not player.start_mpv():
        return

    ws = await establish_connection(api_key)
    if not ws or not await start_task(ws):
        player.stop()
        return

    try:
        wait_time = await continue_task_with_stream_play(ws, text, player)
        await asyncio.sleep(wait_time)
    finally:
        player.stop()
        await close_connection(ws)


if __name__ == "__main__":
    asyncio.run(main())
```

### 事件摘要

| 方向 | `event` | 作用 |
| :--- | :--- | :--- |
| 下行 | `connected_success` | 连接成功 |
| 上行 | `task_start` | 开始任务（含 `model`、`voice_setting`、`audio_setting`） |
| 下行 | `task_started` | 任务已启动 |
| 上行 | `task_continue` | 发送待合成文本片段 |
| 下行 | 含 `data.audio`（hex） | 音频分片 |
| 下行 | `is_final: true` | 本轮结束 |
| 上行 | `task_finish` | 结束会话 |
